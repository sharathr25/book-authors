const request = require('supertest');
const session = require('supertest-session');
 
describe('loading express', () => {
  process.env.DATABASE = 'test';
  process.env.PORT = 5000;
  let server;
  let testSession;
  let authenticatedSession;
  beforeEach(() => {
    // eslint-disable-next-line global-require
    server = require('../server.js');
    testSession = session(server);
  });
  beforeEach((done) => {
    testSession.post('/setsession')
      .send('email=abcd@gmail.com')
      .send('password=1234')
      .expect(200)
      .end(() => {
        authenticatedSession = testSession;
        return done();
      });
  });
  afterEach(() => {
    server.close();
  });
  it('responds to / -status 200', (done) => {
    request(server)
      .get('/')
      .expect(200, done)
      .end((err, res) => {
        console.log(res.text);
        done();
      });
  });
  it('responds to /signup', (done) => {
    request(server)
      .get('/signup')
      .expect(200, done)
      .end((err, res) => {
        console.log(res.text);
        done();
      });
  });
  it('responds to /addUser', (done) => {
    request(server)
      .post('/addUser')
      .send('username=abcd')
      .send('email=abcd@gmail.com')
      .send('password=1234')
      .send('password1=1234')
      .expect(200)
      .end((err, res) => {
        console.log(res.text);
        done();
      });
  });
  it('responds to /login', (done) => {
    request(server)
      .get('/login')
      .expect(200, done)
      .end((err, res) => {
        console.log(res.text);
        done();
      });
  });
  it('responds to /books -status 200', (done) => {
    authenticatedSession.get('/books')
      .expect(200)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('responds to /authors -status 200', (done) => {
    authenticatedSession.get('/authors')
      .expect(200)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('responds to /bookisbn/valid isbn -status 200', (done) => {
    authenticatedSession.get('/bookisbn/123')
      .expect(200)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('responds to /author/valid author id -status 200', (done) => {
    authenticatedSession.get('/author/1')
      .expect(200)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('responds to /bookisbn/invalid isbn -status 500', (done) => {
    authenticatedSession.get('/bookisbn/100')
      .expect(404)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('responds to /author/invalid author id -status 500', (done) => {
    authenticatedSession.get('/author/100')
      .expect(404)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
  it('404 everything else', (done) => {
    request(server)
      .get('/foo/bar')
      .expect(404, done);
  });
  it('responds to /logout', (done) => {
    request(server)
      .get('/logout')
      .expect(200, done);
  });
  it('responds to /books wihtout loggin in', (done) => {
    request(server)
      .get('/books')
      .expect(403)
      .end((err, res) => {
        console.log(res.text);
        console.log('----------------------------------------------');
        done();
      });
  });
});
